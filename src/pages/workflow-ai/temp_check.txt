// Markdown Renderer
const renderMarkdown = (text) => {
    if (!text) return "";
    
    // 1. Limpieza de espacios profundos
    let content = text.trim().replace(/\n{3,}/g, '\n\n');

    // 2. Procesar bloques de cÃ³digo primero (SVG y Otros)
    content = content.replace(/[ \t]*```(svg|csv|[\w]*)\n?([\s\S]*?)```/gi, (match, lang, code) => {
        const cleanLang = lang.toLowerCase().trim();
        const cleanCode = code.trim();

        if (cleanLang === 'svg') {
            return `<div class="my-6 flex justify-center bg-zinc-900/30 p-8 rounded-[2rem] border border-zinc-800/50 shadow-2xl overflow-hidden backdrop-blur-sm">
                ${cleanCode}
            </div>`;
        }

        if (cleanLang === 'csv') {
            const encodedCsv = encodeURIComponent(cleanCode);
            return `<div class="bg-zinc-950/80 border border-zinc-800 rounded-2xl p-4 my-6 backdrop-blur-md">
                <div class="flex items-center justify-between mb-3 border-b border-zinc-800 pb-2">
                    <span class="text-[9px] font-black text-emerald-500 uppercase tracking-[0.2em] flex items-center gap-2">
                        <div class="w-1 h-1 bg-emerald-500 rounded-full animate-pulse shadow-[0_0_5px_#10b981]"></div> Data Source
                    </span>
                    <a href="data:text/csv;charset=utf-8,${encodedCsv}" download="artories_data.csv" class="text-[9px] bg-emerald-500/10 text-emerald-400 hover:bg-emerald-500 hover:text-white px-3 py-1 rounded-full transition-all font-black uppercase tracking-tighter shadow-lg">Download CSV</a>
                </div>
                <pre class="text-[10px] font-mono text-emerald-500/60 overflow-x-auto custom-scrollbar">${cleanCode}</pre>
            </div>`;
        }

        return `<pre class="bg-black border border-zinc-800 rounded-xl p-4 my-4 overflow-x-auto text-[11px] font-mono text-zinc-400 whitespace-pre-wrap">${cleanCode}</pre>`;
    });
